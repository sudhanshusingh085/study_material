Patch & Label & Annotate mutate
bash
# Patch (strategic merge)
oc patch deployment my-app -p '{"spec":{"replicas":3}}'
oc patch deployment my-app --type=json -p '[{"op":"replace","path":"/spec/replicas","value":3}]'
oc patch svc my-svc --type=merge -p '{"spec":{"type":"NodePort"}}'

# Patch image (very common in CD)
oc patch deployment my-app -p '{"spec":{"template":{"spec":{"containers":[{"name":"app","image":"my-reg/app:v2"}]}}}}'

# Set image (shortcut)
oc set image deployment/my-app app=my-reg/app:v2
oc set image deploy/my-app *=my-reg/app:v2      # all containers

# Labels

oc label pod my-pod env-                         # remove label
oc label pods -l app=web tier=frontend
oc label node worker-1 gpu=true
oc label ns my-ns team=backend

# Annotations
oc annotate deployment my-app deployment.kubernetes.io/revision="2"
oc annotate pod my-pod note-                     # remove annotation

# Taint nodes
oc adm taint node worker-1 key=value:NoSchedule
oc adm taint node worker-1 key:NoSchedule-       # remove taint
Output Formats -o
Tip: Combine -o with pipes for powerful filtering. All formats below work with any resource.
bash
# Standard output formats
oc get pods -o wide                              # extra columns (node, IP)
oc get pods -o yaml                              # full yaml
oc get pods -o json                              # full json
oc get pods -o name                              # resource/name format

# Export clean yaml (remove cluster-specific fields)
oc get deployment my-app -o yaml | \
  yq 'del(.metadata.resourceVersion, .metadata.uid, .metadata.creationTimestamp, .status)'

# Get specific field with -o go-template
oc get pod my-pod -o go-template='{{.status.podIP}}'
oc get pods -o go-template='{{range .items}}{{.metadata.name}}{{"\n"}}{{end}}'
JSONPath Queries -o jsonpath
bash
# Single pod fields
oc get pod my-pod -o jsonpath='{.status.podIP}'
oc get pod my-pod -o jsonpath='{.spec.nodeName}'
oc get pod my-pod -o jsonpath='{.status.phase}'
oc get pod my-pod -o jsonpath='{.spec.containers[0].image}'
oc get pod my-pod -o jsonpath='{.spec.containers[*].name}'

# Multiple fields with newline
oc get pod my-pod -o jsonpath='{.metadata.name}{"\t"}{.status.podIP}{"\n"}'

# List all pods: name + status + node
oc get pods -o jsonpath='{range .items[*]}{.metadata.name}{"\t"}{.status.phase}{"\t"}{.spec.nodeName}{"\n"}{end}'

# All pod images
oc get pods -o jsonpath='{range .items[*]}{.metadata.name}{"\t"}{.spec.containers[*].image}{"\n"}{end}'

# Get a specific secret value (base64 decoded)
oc get secret my-secret -o jsonpath='{.data.password}' | base64 -d

# Get service ClusterIP
oc get svc my-svc -o jsonpath='{.spec.clusterIP}'

# Get all node names
oc get nodes -o jsonpath='{.items[*].metadata.name}'

# Nodes with capacity CPU
oc get nodes -o jsonpath='{range .items[*]}{.metadata.name}{"\t"}{.status.capacity.cpu}{"\n"}{end}'

# Deployment ready replicas
oc get deploy my-app -o jsonpath='{.status.readyReplicas}'

# Filter with @.field (conditional)
oc get pods -o json | jq '.items[] | select(.status.phase=="Running") | .metadata.name'
oc get pods -o json | jq '.items[] | select(.spec.nodeName=="worker-1") | .metadata.name'
oc get pods -o json | jq -r '.items[] | [.metadata.name, .status.podIP, .spec.nodeName] | @tsv'

Bash Pipes & Filtering grep · awk · sed · jq
bash
## ─── GREP ───────────────────────────────────────────────────────
# Find pods by partial name
oc get pods | grep "my-app"
oc get pods | grep -v Running                   # NOT running
oc get pods | grep -E "(Error|CrashLoop|OOMKill)"

# Events for specific resource
oc describe pod my-pod | grep -A5 "Events:"

## ─── AWK ─────────────────────────────────────────────────────────
# Print only pod names (column 1)
oc get pods | awk '{print $1}'
# Print name + status (col 1 and 3), skip header
oc get pods | awk 'NR>1 {print $1, $3}'
# Pods with > 1 restart
oc get pods | awk 'NR>1 && $4 > 1 {print $1, $4}'
# Get just names of non-running pods
oc get pods | awk 'NR>1 && $3 != "Running" {print $1}'
# Delete all non-running pods
oc get pods | awk 'NR>1 && $3 != "Running" {print $1}' | xargs oc delete pod

## ─── SED ─────────────────────────────────────────────────────────
# Strip header line
oc get pods | sed '1d'
# Get only the image tag from yaml
oc get deploy my-app -o yaml | grep "image:" | sed 's/.*image: //'

## ─── JQ ──────────────────────────────────────────────────────────
# All pod names + IPs in table form
oc get pods -o json | jq -r '.items[] | [.metadata.name, .status.podIP] | @tsv'
# Only running pods
oc get pods -o json | jq -r '.items[] | select(.status.phase=="Running") | .metadata.name'
# Node name + memory allocatable
oc get nodes -o json | jq -r '.items[] | [.metadata.name, .status.allocatable.memory] | @tsv'
# All secrets names in namespace
oc get secrets -o json | jq -r '.items[].metadata.name'
# Get specific env var value from a deployment
oc get deploy my-app -o json | jq -r '.spec.template.spec.containers[0].env[] | select(.name=="DB_HOST") | .value'

## ─── XARGS ───────────────────────────────────────────────────────
# Restart all pods in a deployment (by deleting them)
oc get pods -l app=my-app -o name | xargs oc delete
# Exec command on all matching pods
oc get pods -l app=my-app -o name | xargs -I{} oc exec {} -- ls /app

## ─── LOOP ────────────────────────────────────────────────────────
# Print logs for all crashlooping pods
for pod in $(oc get pods | awk '/CrashLoop/{print $1}'); do
  echo "=== $pod ==="
  oc logs $pod --previous --tail=30
done
PowerShell Pipes & Filtering PS
powershell
# Select-String (like grep)
oc get pods | Select-String "my-app"
oc get pods | Select-String -NotMatch "Running"
oc get pods | Select-String -Pattern "Error|CrashLoop"

# ConvertFrom-Json + Where-Object
$pods = oc get pods -o json | ConvertFrom-Json
$pods.items | Where-Object { $_.status.phase -eq "Running" } |
  Select-Object -ExpandProperty metadata | Select-Object name

# Get pod name + IP + node (table)
$pods = oc get pods -o json | ConvertFrom-Json
$pods.items | ForEach-Object {
  [PSCustomObject]@{
    Name   = $_.metadata.name
    Phase  = $_.status.phase
    PodIP  = $_.status.podIP
    Node   = $_.spec.nodeName
  }
} | Format-Table -AutoSize

# Find pods with restarts > 0
$pods.items | Where-Object {
  $_.status.containerStatuses -and $_.status.containerStatuses[0].restartCount -gt 0
} | ForEach-Object {
  [PSCustomObject]@{ Name = $_.metadata.name; Restarts = $_.status.containerStatuses[0].restartCount }
} | Sort-Object Restarts -Descending

# Get specific column from text output (by index)
$output = (oc get pods) -split "`n" | Select-Object -Skip 1
$output | ForEach-Object {
  $cols = $_ -split '\s+'
  [PSCustomObject]@{ Name = $cols[0]; Status = $cols[2]; Restarts = $cols[3] }
} | Format-Table

# Decode secret value
$secret = oc get secret my-secret -o json | ConvertFrom-Json
[System.Text.Encoding]::UTF8.GetString([Convert]::FromBase64String($secret.data.password))

# Get all deployments with image tags
$deps = oc get deployments -o json | ConvertFrom-Json
$deps.items | ForEach-Object {
  [PSCustomObject]@{
    Deployment = $_.metadata.name
    Image      = $_.spec.template.spec.containers[0].image
    Replicas   = $_.spec.replicas
    Ready      = $_.status.readyReplicas
  }
} | Format-Table -AutoSize

# Delete all non-running pods
$pods.items | Where-Object { $_.status.phase -ne "Running" } |
  ForEach-Object { oc delete pod $_.metadata.name }

Volumes — All Types spec.volumes
Remember: Volumes are declared at spec.volumes (pod level). They are mounted into containers via spec.containers[].volumeMounts.
All volume types
spec:
  volumes:

    # ── 1. emptyDir — ephemeral scratch space ────────────────────────
    - name: scratch
      emptyDir: {}

    # emptyDir in memory (RAM-backed, disappears with pod)
    - name: cache
      emptyDir:
        medium: Memory
        sizeLimit: 256Mi

    # ── 2. ConfigMap ─────────────────────────────────────────────────
    - name: app-config-vol
      configMap:
        name: app-config
        defaultMode: 0644

    # ConfigMap — mount only specific keys as files
    - name: nginx-config-vol
      configMap:
        name: nginx-config
        items:
          - key: nginx.conf       # CM key
            path: nginx.conf     # filename in mount dir
          - key: default.conf
            path: conf.d/default.conf

    # ── 3. Secret ────────────────────────────────────────────────────
    - name: app-secret-vol
      secret:
        secretName: app-secret
        defaultMode: 0400

    # Secret — mount specific key as file
    - name: tls-vol
      secret:
        secretName: tls-secret
        items:
          - key: tls.crt
            path: tls.crt
          - key: tls.key
            path: tls.key

    # ── 4. PersistentVolumeClaim ─────────────────────────────────────
    - name: data-vol
      persistentVolumeClaim:
        claimName: my-pvc
        readOnly: false

    # ── 5. hostPath — node filesystem ────────────────────────────────
    - name: host-logs
      hostPath:
        path: /var/log
        type: Directory   # DirectoryOrCreate | FileOrCreate | File | Socket | CharDevice | BlockDevice

    # ── 6. Projected — combine multiple sources ───────────────────────
    - name: projected-vol
      projected:
        sources:
          - configMap:
              name: app-config
          - secret:
              name: app-secret
          - serviceAccountToken:
              path: token
              expirationSeconds: 3600
              audience: api
          - downwardAPI:
              items:
                - path: labels
                  fieldRef:
                    fieldPath: metadata.labels

    # ── 7. NFS ───────────────────────────────────────────────────────
    - name: nfs-vol
      nfs:
        server: nfs-server.example.com
        path: /exports/data
        readOnly: false

    # ── 8. downwardAPI — expose pod metadata as files ────────────────
    - name: podinfo
      downwardAPI:
        items:
          - path: name
            fieldRef:
              fieldPath: metadata.name
          - path: namespace
            fieldRef:
              fieldPath: metadata.namespace
          - path: labels
            fieldRef:
              fieldPath: metadata.labels
          - path: cpu_limit
            resourceFieldRef:
              containerName: app
              resource: limits.cpu

    # ── 9. CSI (generic CSI driver - e.g. Secrets Store) ─────────────
    - name: secrets-store
      csi:
        driver: secrets-store.csi.k8s.io
        readOnly: true
        volumeAttributes:
          secretProviderClass: my-provider

    # ── 10. configMap as env file ─────────────────────────────────────
    - name: config-file
      configMap:
        name: app-properties
        defaultMode: 0755        # executable
VolumeMounts containers[].volumeMounts
yaml
containers:
  - name: app
    volumeMounts:

      # Mount entire volume to path
      - name: data-vol              # must match spec.volumes[].name
        mountPath: /data

      # ReadOnly mount (secrets, certs)
      - name: tls-vol
        mountPath: /etc/tls
        readOnly: true

      # Mount specific subPath (single file, not whole dir)
      - name: nginx-config-vol
        mountPath: /etc/nginx/nginx.conf
        subPath: nginx.conf        # file inside volume
        readOnly: true

      # Mount config file using subPath (no overwrite of whole dir)
      - name: app-config-vol
        mountPath: /app/config/application.properties
        subPath: application.properties

      # subPathExpr (use env var in subPath)
      - name: data-vol
        mountPath: /data/$(POD_NAME)
        subPathExpr: $(POD_NAME)   # uses env var POD_NAME

      # Scratch space
      - name: scratch
        mountPath: /tmp

      # Downward API files
      - name: podinfo
        mountPath: /etc/podinfo
        readOnly: true

      # mountPropagation - shared volume (advanced)
      - name: shared
        mountPath: /shared
        mountPropagation: HostToContainer  # None | HostToContainer | Bidirectional

Resources & Limits cpu · memory
yaml
containers:
  - name: app
    resources:
      requests:                     # scheduler uses this to place pod on node
        cpu: "250m"                  # 250 millicores = 0.25 CPU
        memory: "256Mi"
        ephemeral-storage: "1Gi"
      limits:                       # hard cap: OOMKill on memory, throttle on CPU
        cpu: "500m"
        memory: "512Mi"
        ephemeral-storage: "2Gi"
        # GPU (NVIDIA operator)
        nvidia.com/gpu: 1

# CPU units: 1 = 1000m. 1m = 0.001 CPU core
# Memory units: Ki, Mi, Gi, Ti (power of 2) or K, M, G (power of 10)
# QoS Classes:
#   Guaranteed  = requests == limits (best for latency-sensitive)
#   Burstable   = requests < limits
#   BestEffort  = no requests or limits (evicted first)
Probes liveness · readiness · startup
All probe types
containers:
  - name: app

    # ── Startup Probe (checked first, disables liveness until passes) ─
    startupProbe:
      httpGet:
        path: /health/startup
        port: 8080
      failureThreshold: 30          # 30 * 10s = 5 min max startup time
      periodSeconds: 10

    # ── Liveness Probe (restart if fails) ────────────────────────────
    livenessProbe:
      # HTTP check
      httpGet:
        path: /actuator/health
        port: 8080
        scheme: HTTP                # or HTTPS
        httpHeaders:
          - name: X-Health-Check
            value: "true"
      initialDelaySeconds: 30
      periodSeconds: 10
      timeoutSeconds: 5
      failureThreshold: 3
      successThreshold: 1

    # ── Readiness Probe (remove from Service endpoints if fails) ─────
    readinessProbe:
      # TCP socket check
      tcpSocket:
        port: 8080
      initialDelaySeconds: 5
      periodSeconds: 5

    # ── Exec probe (run command inside container) ────────────────────
    livenessProbe:
      exec:
        command:
          - cat
          - /tmp/healthy
      initialDelaySeconds: 5
      periodSeconds: 5

    # ── gRPC probe (k8s 1.23+) ───────────────────────────────────────
    livenessProbe:
      grpc:
        port: 9090
        service: my.service.Health
      periodSeconds: 10

Affinity, Taints & Tolerations scheduling
yaml
spec:
  # ── nodeSelector (simple) ─────────────────────────────────────────
  nodeSelector:
    kubernetes.io/arch: amd64
    gpu: "true"

  # ── Affinity ──────────────────────────────────────────────────────
  affinity:
    # Node affinity
    nodeAffinity:
      requiredDuringSchedulingIgnoredDuringExecution:
        nodeSelectorTerms:
          - matchExpressions:
              - key: node-role.kubernetes.io/worker
                operator: Exists   # In | NotIn | Exists | DoesNotExist | Gt | Lt
              - key: zone
                operator: In
                values: [zone-a, zone-b]
      preferredDuringSchedulingIgnoredDuringExecution:
        - weight: 100
          preference:
            matchExpressions:
              - key: ssd
                operator: Exists

    # Pod anti-affinity (spread pods across nodes)
    podAntiAffinity:
      requiredDuringSchedulingIgnoredDuringExecution:
        - labelSelector:
            matchExpressions:
              - key: app
                operator: In
                values: [my-app]
          topologyKey: kubernetes.io/hostname  # one pod per node

    # Pod affinity (co-locate with another app)
    podAffinity:
      preferredDuringSchedulingIgnoredDuringExecution:
        - weight: 50
          podAffinityTerm:
            labelSelector:
              matchLabels:
                app: cache
            topologyKey: kubernetes.io/hostname

  # ── Tolerations (allow scheduling on tainted nodes) ──────────────
  tolerations:
    - key: gpu
      operator: Exists
      effect: NoSchedule
    - key: dedicated
      value: ml-workloads
      operator: Equal
      effect: NoSchedule             # NoSchedule | PreferNoSchedule | NoExecute
    - key: node.kubernetes.io/not-ready
      operator: Exists
      effect: NoExecute
      tolerationSeconds: 300         # evict after 300s if still not ready

  # ── TopologySpreadConstraints (spread evenly) ────────────────────
  topologySpreadConstraints:
    - maxSkew: 1
      topologyKey: topology.kubernetes.io/zone
      whenUnsatisfiable: DoNotSchedule  # DoNotSchedule | ScheduleAnyway
      labelSelector:
        matchLabels:
          app: my-app

# ── Role (namespace-scoped) ───────────────────────────────────────
apiVersion: rbac.authorization.k8s.io/v1
kind: Role
metadata:
  name: pod-reader
  namespace: my-ns
rules:
  - apiGroups: [""]               # "" = core API group
    resources: [pods, pods/log]
    verbs: [get, list, watch]
  - apiGroups: [apps]
    resources: [deployments]
    verbs: [get, list, watch, update, patch]
  - apiGroups: [""]
    resources: [configmaps]
    resourceNames: [app-config]   # restrict to specific resource
    verbs: [get]
---

apiVersion: rbac.authorization.k8s.io/v1
kind: RoleBinding
metadata:
  name: pod-reader-binding
  namespace: my-ns
subjects:
  - kind: ServiceAccount
    name: my-sa
    namespace: my-ns
  - kind: User
    name: user@example.com
  - kind: Group
    name: dev-team
roleRef:
  kind: Role
  name: pod-reader
